BankSimul REST API dokumentaatio

Asiakas

GET /asiakas = Hae kaikki asiakkaat

GET /asiakas/:id = Hae asiakas jonka id on :id

POST /asiakas = Lisää asiakas

PUT /asiakas/:id = Päivitä asiakas jonka id on :id

DELETE /asiakas/:id = Poista asiakas jonka id on :id

Asiakkaan tietorakenne:

x-www-form-urlencoded

{

"tunnus": "JAKY001",

"etunimi": "Jarkko",

"sukunimi": "Kylähaara",

"osoite": "Mansikkapolku 1 04444 Vaasa",

"puhelin": "050-2294966"

}

Tili

GET /tili = Hae kaikki tilit

GET /tili/:id = Hae tili jonka id on :id

POST /tili = Lisää tili

PUT /tili/:id = Päivitä tili jonka id on :id

DELETE /tili/:id = Poista tili jonka id on :id

Tilin tietorakenne:

x-www-form-urlencoded

{

"tilinumero": "123456789012345678",

"saldo": "100000"

}
